Функция ПечатьАкта(МассивОбъектов, ОбъектыПечати) Экспорт
	
	//параметры
	ТабДок = Новый ТабличныйДокумент;
	ТабДок.ОриентацияСтраницы = ОриентацияСтраницы.Портрет;
	ТабДок.АвтоМасштаб = Истина;
	ТабДок.ИмяПараметровПечати = "АктУстановкиЗапасныхЧастей";
	
	Макет = ПолучитьМакет("МакетАктУстановкиЗЧ");
	ОбластьШапка = Макет.ПолучитьОбласть("ОбластьШапка");
	ОбластьСтрока = Макет.ПолучитьОбласть("ОбластьСтрока");
	ОбластьПодписи = Макет.ПолучитьОбласть("ОбластьПодписи");
	
	//данные
	ДокументТН=МассивОбъектов[0];
	АвтоНаименование=ПолучитьАвтоНаименование(ДокументТН.Автомобиль);
	
	//параметры
	СтруктураОбластьЛист = Новый Структура;
	СтруктураОбластьЛист.Вставить("ДокДата", Формат(ДокументТН.Дата,"ДЛФ=DD"));
	СтруктураОбластьЛист.Вставить("АвтоНаименование", АвтоНаименование);
	
	//шапка
	ОбластьШапка.Параметры.Заполнить(СтруктураОбластьЛист);
	ТабДок.Вывести(ОбластьШапка);
	
	//строки
	Для Каждого СтрМ Из ДокументТН.Материалы Цикл
		СтруктураОбластьСтрока = Новый Структура;
		СтруктураОбластьСтрока.Вставить("НомНаименование", СокрЛП(СтрМ.Номенклатура.Наименование));
		СтруктураОбластьСтрока.Вставить("ЕдИзм", СокрЛП(СтрМ.Номенклатура.ЕдиницаИзмерения));
		СтруктураОбластьСтрока.Вставить("Колво", СтрМ.Количество);
		ОбластьСтрока.Параметры.Заполнить(СтруктураОбластьСтрока);
		ТабДок.Вывести(ОбластьСтрока);
	КонецЦикла;
	
	//подписи
	ОбластьПодписи.Параметры.Заполнить(СтруктураОбластьЛист);
	ТабДок.Вывести(ОбластьПодписи);
	
	//завершение
	Возврат ТабДок;
	
КонецФункции
