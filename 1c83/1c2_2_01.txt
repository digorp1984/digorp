//создание документов по истории ОС
Процедура СозданиеДокументовПоИсторииОС(СтрОперации)
	
	Если СокрЛП(СтрОперации.КодОперации)="A" Тогда
		//ввод истории переоценки
		ДокПереоценка=Документы.ОС_ВводИсторииПереоценки.СоздатьДокумент();
		ДокПереоценка.Дата=СтрОперации.ДатаОперации;
		ДокПереоценка.Учреждение=Учреждение;
		ДокПереоценка.ОС=СтрОперации.ОССсылка;
		ДокПереоценка.Комментарий="Загружено автоматически";
		ТЧПереоценка=ДокПереоценка.Переоценка.Добавить();
		ТЧПереоценка.ПереоценкаПоСостояниюНА=СтрОперации.ДатаОперации;
		ТЧПереоценка.ОсновнойКоэффициент=СтрОперации.Коэфициент;
		ТЧПереоценка.ДооценкаБалансовойСтоимости=СтрОперации.СуммаБаланаса1;
		ТЧПереоценка.БалансоваяСтоимостьПослеПереоценки=СтрОперации.СуммаБаланаса1+СтрОперации.СуммаБаланаса2;
		ТЧПереоценка.ДооценкаАмортизации=СтрОперации.СуммаАмортизации1;
		ТЧПереоценка.АмортизацияПослеПереоценки=СтрОперации.СуммаАмортизации1+СтрОперации.СуммаАмортизации2;
		ТЧПереоценка.ДатаПроведения=СтрОперации.ДатаОперации;
		ДокПереоценка.Записать(РежимЗаписиДокумента.Запись);
		ДокПереоценка.Записать(РежимЗаписиДокумента.Проведение);
		
	ИначеЕсли СокрЛП(СтрОперации.КодОперации)="B" Тогда
		//ввод истории капремонта
		ДокКапремонт=Документы.ОС_ВводИсторииКапремонта.СоздатьДокумент();
		ДокКапремонт.Дата=СтрОперации.ДатаОперации;
		ДокКапремонт.Учреждение=Учреждение;
		ДокКапремонт.ОС=СтрОперации.ОССсылка;
		ДокКапремонт.ИнвНомер=СтрОперации.ИнвНомерСсылка;
		ДокКапремонт.Комментарий="Загружено автоматически";
		ТЧКапремонт=ДокКапремонт.Капремонт.Добавить();
		ТЧКапремонт.ДатаКапремонта=СтрОперации.ДатаОперации;
		ТЧКапремонт.НомерДокументаОКапРемонте="-";
		ТЧКапремонт.СуммаНаКапРемонт=СтрОперации.СуммаБаланаса1;
		ДокКапремонт.Записать(РежимЗаписиДокумента.Запись);
		ДокКапремонт.Записать(РежимЗаписиДокумента.Проведение);
		
	ИначеЕсли СокрЛП(СтрОперации.КодОперации)="D" Тогда
		//ввод истории модернизации
		ДокКапремонт=Документы.ОС_ВводИсторииКапремонта.СоздатьДокумент();
		ДокКапремонт.Дата=СтрОперации.ДатаОперации;
		ДокКапремонт.Учреждение=Учреждение;
		ДокКапремонт.ОС=СтрОперации.ОССсылка;
		ДокКапремонт.ИнвНомер=СтрОперации.ИнвНомерСсылка;
		ДокКапремонт.Комментарий="Загружено автоматически";
		ТЧМодернизация=ДокКапремонт.Модернизация.Добавить();
		ТЧМодернизация.ДатаМодернизации=СтрОперации.ДатаОперации;
		ТЧМодернизация.НомерДокументаМодернизации="-";
		ТЧМодернизация.СуммаМодерн=СтрОперации.СуммаБаланаса1;
		ТЧМодернизация.СрокПолезногоИспользованияПослеМодернизации=СтрОперации.СрокПолезногоИспользования;
		ДокКапремонт.Записать(РежимЗаписиДокумента.Запись);
		ДокКапремонт.Записать(РежимЗаписиДокумента.Проведение);
		
	ИначеЕсли СокрЛП(СтрОперации.КодОперации)="С" Тогда
		//ввод истории амортизации
		ДокАморт=Документы.ОС_ВводИсторииАмортизации.СоздатьДокумент();
		ДокАморт.Дата=СтрОперации.ДатаОперации;
		ДокАморт.Учреждение=Учреждение;
		ДокАморт.ОС=СтрОперации.ОССсылка;
		ДокАморт.ИнвНомер=СтрОперации.ИнвНомерСсылка;
		ДокАморт.Комментарий="Загружено автоматически";
		ТЧДокАморт=ДокАморт.ОсновныеСредства.Добавить();
		ТЧДокАморт.ДатаНачисленияАмортизации=СтрОперации.ДатаОперации;
		ТЧДокАморт.СуммаАмортизации=СтрОперации.СуммаАмортизации;
		ДокАморт.Записать(РежимЗаписиДокумента.Запись);
		ДокАморт.Записать(РежимЗаписиДокумента.Проведение);
	КонецЕсли;
	
КонецПроцедуры
