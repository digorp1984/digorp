//поиск или создание ОС по забалансу
Функция ОпределитьОСзаб(КФО,Счет,МОЛ,Наименование)
	
	//общие параметры
	СпрОС=Справочники.ОсновныеСредства;
	//НаименованиеОС=Наименование+" ("+СокрЛП(КФО)+"."+Прав(СокрЛП(МОЛ.Код),4)+")";
	НаименованиеОС=СокрЛП(Лев(Наименование,75))+"["+Прав(СокрЛП(МОЛ.Код),3)+"]";
	
	//определение ОС
	ОС=СпрОС.НайтиПоНаименованию(НаименованиеОС);
	Если ОС=СпрОС.ПустаяСсылка() Тогда
		НовЭлемент=СпрОС.СоздатьЭлемент();
		НовЭлемент.Учреждение=Учреждение;
		НовЭлемент.Родитель=ГруппаОС;
		НовЭлемент.Наименование=НаименованиеОС;
		НовЭлемент.НаименованиеПолное=НаименованиеОС;
		НовЭлемент.СчетУчета=Счет;
		НовЭлемент.КБК=КБКПоУмолчанию;
		НовЭлемент.КВД=КФО;
		НовЭлемент.ВидНФА=Перечисления.ВидыНФА.ОС;
		НовЭлемент.ВидУчета=Перечисления.ВидыУчетаНФА.НаБалансе;
		НовЭлемент.ИсточникПоступления=Перечисления.ИсточникиПоступления.ПриобретенноеЗаСчетДоходов;
		НовЭлемент.ВидИмущества=Перечисления.ВидыИмущества.ПрочееДвижимое;
		НовЭлемент.ГрупповойУчет=Истина;
		НовЭлемент.Записать();
		
		Сообщить("    создано ОС (забланс): "+СокрЛП(НовЭлемент.Ссылка));
		ОС=НовЭлемент.Ссылка;
	КонецЕсли;
	
	//завершение
	Возврат ОС;
	
КонецФункции
